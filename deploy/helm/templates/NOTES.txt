ContextStream has been deployed!

1. Get the app URL:
{{- if .Values.ingress.enabled }}
  http{{ if .Values.ingress.tls }}s{{ end }}://{{ (first .Values.ingress.hosts).host }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get svc {{ include "contextstream.fullname" . }} --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "http://$NODE_IP:$NODE_PORT"
{{- else }}
  kubectl port-forward svc/{{ include "contextstream.fullname" . }} 3000:{{ .Values.service.port }} --namespace {{ .Release.Namespace }}
  Then open: http://localhost:3000
{{- end }}

2. Run database migrations:
  kubectl exec -it deploy/{{ include "contextstream.fullname" . }}-app --namespace {{ .Release.Namespace }} -- npm run db:migrate

3. Seed initial data (optional):
  kubectl exec -it deploy/{{ include "contextstream.fullname" . }}-app --namespace {{ .Release.Namespace }} -- npm run db:seed

4. Check pod status:
  kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}

{{- if .Values.worker.enabled }}

5. Worker is enabled. To check worker logs:
  kubectl logs -f deploy/{{ include "contextstream.fullname" . }}-worker --namespace {{ .Release.Namespace }}
{{- end }}

{{- if .Values.pdfParser.enabled }}

6. PDF parser is enabled at internal URL:
  http://{{ include "contextstream.fullname" . }}-pdf-parser:8000
{{- end }}
